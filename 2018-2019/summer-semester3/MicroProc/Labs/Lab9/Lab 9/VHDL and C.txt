#include <stdio.h>
#include "system.h"
#include "altera_avalon_pio_regs.h"
int main()
{
	int count=0x0001;
	//int shiftflag = 1;
	int delay;
	printf("Hello from Nios II!\n");

	while(1)
	{

		IOWR_ALTERA_AVALON_PIO_DATA(PIO_LED_BASE, count);
		for(delay=0;delay<100;delay++);
		count= count+1;
	}
	return 0;
}
